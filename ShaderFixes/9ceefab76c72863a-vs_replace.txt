void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	o0.xy = mul(v0, sf.pos).xy;
	o0.zw = v0.zw;
	o1.xy = v1.xy;
		
	if (!rt_res_size_match()) // Render Target
		return;
	float layer = 1;
		
	if (filter_index > 0)
		layer = filter_index;
	
	gui_element(o0,layer,0.002);
	
	return;
}
