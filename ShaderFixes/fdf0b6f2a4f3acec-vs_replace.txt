void main(
	int4 v0 : POSITION0,
	float4 v1 : COLOR0,
	float4 v2 : COLOR1,
	out float4 o0 : SV_Position0,
	out float4 o1 : COLOR0,
	out float4 o2 : COLOR1)
{
	float4 r0;
	
	r0 = (int4)v0;
	o0 = mul(r0, sf.pos);
	o1 = v1.zyxw;
	o2 = v2.zyxw;
	
	stereo_unfix(o0);
	
	float layer = 3.5;
	if (texture_filter > 0)
		layer = texture_filter;
	
	gui_element(o0,layer,50);
	
	return;
}
