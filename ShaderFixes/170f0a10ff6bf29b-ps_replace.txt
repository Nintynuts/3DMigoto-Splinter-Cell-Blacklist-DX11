// Ambient Occlusion Character

cbuffer CB_PerFrame : register(b13)
{
	float4 gCameraFadeAlpha : packoffset(c0);
	float4 gCameraFadeShadow : packoffset(c1);
	float4 gColorControl : packoffset(c2);
	float3 cBoostCol : packoffset(c3);
	float cHdrControl : packoffset(c3.w);
	float gAOControl : packoffset(c4);
}

cbuffer CB_PS_AOSphere : register(b7)
{
	float4x3 cScreenToWorld : packoffset(c0);
	float2 cIDFactor : packoffset(c3);
	float cAlphaTest : packoffset(c3.z);
}

SamplerState sIDMap_s : register(s0);
SamplerState sNormalMap_s : register(s1);
SamplerState sAOSphereMap_s : register(s3);
Texture2D<float4> sIDMap : register(t0);
Texture2D<float4> sNormalMap : register(t1);
Texture2D<float4> sAOSphereMap : register(t2);

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : SV_Position0,
	float4 v1 : TEXCOORD0,
	float v2 : TEXCOORD1,
	out float4 o0 : SV_Target0)
{
	float4 r0,r1,r2,r3,r4;
	
	r0.xyzw = v1.xyzw / v2.xxxx;
	r1.z = sIDMap.Sample(sIDMap_s, r0.zw).x;
	r1.xy = r1.zz * r0.xy;
	r0.xyz = sNormalMap.Sample(sNormalMap_s, r0.zw).xyz;
	r0.xyz = r0.yzx * float3(2,2,2) + float3(-1,-1,-1);
	r1.w = 1;
			
	stereo_unfix(r1);
	
	r2.w = dot(r1.xyzw, cScreenToWorld._m00_m10_m20_m30);
	r2.y = dot(r1.xyzw, cScreenToWorld._m01_m11_m21_m31);
	r2.z = dot(r1.xyzw, cScreenToWorld._m02_m12_m22_m32);
	r0.w = dot(r2.yzw, r2.yzw);
	r1.x = rsqrt(r0.w);
	r0.w = 4 * r0.w;
	r3.x = min(1, r0.w);
	r1.yzw = r2.yzw * r1.xxx;
	r3.y = 0.5;
	r3.x += 0.01;
	r2.xy = sAOSphereMap.SampleLevel(sAOSphereMap_s, r3.xy, 0).xy;
	r1.x = 1;
	r1.xyzw = r1.xyzw * r2.xyyy + float4(-1,-0,-0,-0);
	r1.xyzw = cIDFactor.yyyy * r1.xyzw + float4(1,0,0,0);
	r2.x = dot(r1.xyzw, r1.xyzw);
	r3.xyz = r2.xxx * r1.yzw;
	r4.xyz = r1.xxx * r1.yzw;
	r2.yzw = r4.xyz + r4.xyz;
	r3.yzw = r1.xxx * r2.yzw + r3.xyz;
	r3.x = dot(r2.xyzw, r1.xyzw);
	r0.w = dot(r0.xyz, r0.xyz);
	r0.w = rsqrt(r0.w);
	r0.yzw = r0.xyz * r0.www;
	r0.x = 1;
	r0.x = saturate(dot(r3.xyzw, r0.xyzw));
	o0.xyzw = gColorControl.wwww * r0.xxxx;
	return;
}
