// ShadowNet - Text Batched

cbuffer CB_VS_ScaleformBatched : register(b7)
{
	float4 mvp[192] : packoffset(c0);
}

// 3Dmigoto declarations
#define cmp -
Texture1D<float4> IniParams : register(t120);
Texture2D<float4> StereoParams : register(t125);

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	float4 v2 : COLOR0,
	float4 v3 : COLOR1,
	out float4 o0 : SV_Position0,
	out float4 o1 : TEXCOORD0,
	out float4 o2 : TEXCOORD1,
	out float4 o3 : TEXCOORD2,
	out float4 o4 : COLOR0)
{
	float4 r0,r1;
	
	r0.x = 2040.01001 * v3.x;
	r0.x = (int)r0.x;
	o0.x = dot(v0.xyzw, mvp[r0.x].xyzw);
	r1.xyzw = (int4)r0.xxxx + int4(1,2,3,4);
	r0.x = (int)r0.x + 5;
	o2.xyzw = mvp[r0.x].xyzw;
	o0.y = dot(v0.xyzw, mvp[r1.x].xyzw);
	o0.z = dot(v0.xyzw, mvp[r1.y].xyzw);
	o0.w = dot(v0.xyzw, mvp[r1.z].xyzw);
	o3.xyzw = mvp[r1.w].xyzw;
	o1.xy = v1.xy;
	o4.xyzw = v2.zyxw;
	
	float4 resolution = IniParams.Load(int2(0,0));
	int factor = resolution.x == resolution.z ? 1 : 1;
	
	float4 stereo = StereoParams.Load(0);
	o0.x -= stereo.x * (o0.w - stereo.y)/factor;
	
	float4 tex_filter = IniParams.Load(int2(2,0));
	if (tex_filter.x > 0) {
		o0.x -= stereo.x * (tex_filter.z * tex_filter.x - tex_filter.z*5)*50;
	}
	
	return;
}
