// ShadowNet - Text Batched

#include "CB/VS_ScaleformBatched.hlsl"
#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	float4 v2 : COLOR0,
	float4 v3 : COLOR1,
	out float4 o0 : SV_Position0,
	out float4 o1 : TEXCOORD0,
	out float4 o2 : TEXCOORD1,
	out float4 o3 : TEXCOORD2,
	out float4 o4 : COLOR0)
{	
	int r0 = (int)(2040.01001 * v3.x);
	o0.x = dot(v0, mvp[r0 + 0]);
	o0.y = dot(v0, mvp[r0 + 1]);
	o0.z = dot(v0, mvp[r0 + 2]);
	o0.w = dot(v0, mvp[r0 + 3]);
	o3 = mvp[r0 + 4];
	o2 = mvp[r0 + 5];
	o1.xy = v1.xy;
	o4 = v2.zyxw;
	
	stereo_unfix(o0);

	if (texture_filter > 0)
		gui_element(o0,texture_filter,50);
	
	return;
}
