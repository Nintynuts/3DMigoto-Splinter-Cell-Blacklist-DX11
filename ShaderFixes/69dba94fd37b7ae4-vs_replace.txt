// ShadowNet - Text Batched

cbuffer CB_VS_ScaleformBatched : register(b7)
{
	float4 mvp[192] : packoffset(c0);
}

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	float4 v2 : COLOR0,
	float4 v3 : COLOR1,
	out float4 o0 : SV_Position0,
	out float4 o1 : TEXCOORD0,
	out float4 o2 : TEXCOORD1,
	out float4 o3 : TEXCOORD2,
	out float4 o4 : COLOR0)
{
	float4 r0,r1;
	
	r0.x = 2040.01001 * v3.x;
	r0.x = (int)r0.x;
	o0.x = dot(v0, mvp[r0.x]);
	r1 = (int4)r0.x + int4(1,2,3,4);
	r0.x = (int)r0.x + 5;
	o2 = mvp[r0.x];
	o0.y = dot(v0, mvp[r1.x]);
	o0.z = dot(v0, mvp[r1.y]);
	o0.w = dot(v0, mvp[r1.z]);
	o3 = mvp[r1.w];
	o1.xy = v1.xy;
	o4 = v2.zyxw;
	
	stereo_unfix(o0);

	if (texture_filter > 0)
		gui_element(o0,texture_filter,50);
	
	return;
}
