// Directional

#include "CB_PerCamera.hlsl"

cbuffer CB_PerCubeMapRenderCell : register(b0)
{
	float2 gCubemapScale : packoffset(c0);
}

#define LIGHT_REG b1
#include "CB_PS_PerLightBatch.hlsl"

cbuffer CB_PS_DeferredDirLight : register(b7)
{
	float4x4 cScreenToWorld : packoffset(c0);
}

SamplerState sDepth_s : register(s0);
SamplerState sNormal_s : register(s1);
SamplerState sSceneAlbedo_s : register(s2);
SamplerState sSceneLight_s : register(s3);
SamplerState sShadowSamples_s : register(s15);
Texture2D<float4> sDepth : register(t0);
Texture2D<float4> sNormal : register(t1);
Texture2D<float4> sShadowSamples : register(t2);
Texture2D<float4> sSceneAlbedo : register(t3);
Texture2D<float4> sSceneLight : register(t4);

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : SV_Position0,
	float4 v1 : TEXCOORD0,
	float v2 : TEXCOORD1,
	out float4 o0 : SV_Target0)
{
	float4 r0,r1,r2,r3;

	r0.zw = float2(1,1);
	r1.xyzw = v1.xyzw / v2.xxxx;
	r0.xy = r1.xy;
	r1.xy = sDepth.Sample(sDepth_s, r1.zw).xw;
	r0.xyz = r1.xxx * r0.xyz;

	stereo_unfix(r0);

	r2.x = dot(r0.xyzw, cScreenToWorld._m00_m10_m20_m30);
	r2.y = dot(r0.xyzw, cScreenToWorld._m01_m11_m21_m31);
	r2.z = dot(r0.xyzw, cScreenToWorld._m02_m12_m22_m32);
	r0.xyz = cViewPos.xyz + -r2.xyz;
	r0.w = dot(r0.xyz, r0.xyz);
	r0.w = rsqrt(r0.w);
	r0.xyz = r0.xyz * r0.www + cDir0.dir.xyz;
	r0.w = dot(r0.xyz, r0.xyz);
	r0.w = rsqrt(r0.w);
	r0.xyz = r0.xyz * r0.www;
	r2.xyzw = sNormal.Sample(sNormal_s, r1.zw).xyzw;
	r2.xyz = r2.xyz * float3(2,2,2) + float3(-1,-1,-1);
	r0.w = 11 * r2.w;
	r0.w = exp2(r0.w);
	r0.w = max(9.99999975e-006, r0.w);
	r1.x = dot(r2.xyz, r2.xyz);
	r1.x = rsqrt(r1.x);
	r2.xyz = r2.xyz * r1.xxx;
	r0.x = saturate(dot(r0.xyz, r2.xyz));
	r0.y = dot(r2.xyz, cDir0.dir.xyz);
	r0.x = log2(r0.x);
	r0.x = r0.w * r0.x;
	r0.z = 2 + r0.w;
	r0.x = exp2(r0.x);
	r0.x = r0.z * r0.x;
	r0.z = r0.y * 0.5 + 0.5;
	r0.y = saturate(r0.y);
	r0.z = r0.z * r0.z;
	r0.z = r0.z * r0.z + -r0.y;
	r0.y = r1.y * r0.z + r0.y;
	r0.z = saturate(r0.y);
	r2.xyz = sShadowSamples.Sample(sShadowSamples_s, r1.zw).xyz;
	r2.w = 1;
	r0.w = dot(cDir0.mask.xyzw, r2.xyzw);
	r0.z = r0.w * r0.z;
	r0.xy = r0.zy * r0.xw;
	r2.xyz = cDir0.dCol.xyz * r0.yyy;
	r0.x = 0.125 * r0.x;
	r0.xyz = cDir0.sCol.xyz * r0.xxx;
	r3.xyzw = sSceneLight.Sample(sSceneLight_s, r1.zw).xyzw;
	r1.xyzw = sSceneAlbedo.Sample(sSceneAlbedo_s, r1.zw).xyzw;
	r1.xyzw = r1.xyzw * r1.xyzw;
	r3.xyz = r3.xyz * r3.www;
	r3.xyz = r3.xyz * r3.xyz;
	r3.xyz = gCubemapScale.xxx * r3.xyz;
	r1.xyz = r2.xyz * r1.xyz + r3.xyz;
	r0.w = 2.66666675 * r1.w;
	o0.xyz = r0.xyz * r0.www + r1.xyz;
	o0.w = 1;
	return;
}
