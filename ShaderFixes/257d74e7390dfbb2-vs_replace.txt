#include "CB_VS_Scaleform.hlsl"

#include "3Dmigoto.hlsl"

void main(
	int4 v0 : POSITION0,
	float4 v1 : COLOR0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0,
	out float4 o2 : COLOR0)
{
	float4 r0;
	
	r0 = (int4)v0;
	o0 = mul(r0, sf.pos);
	o1 = mul(r0, sf.tex);
	o2 = v1.zyxw;
	
	stereo_unfix(o0);
	
	float layer = 1;
	if (texture_filter > 0)
		layer = texture_filter;

	gui_element(o0,layer,50);
	
	return;
}
