// ShadowNet - Block colour

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	o0.xyzw = v0.xyzw;
	o1.xy = float2(1,-1) * v0.xy;

	if (menu_open)
	{
		stereo_unfix(o0);
		
		if (texture_filter > 0)
			gui_element(o0,texture_filter,50);
	}
	return;
}
