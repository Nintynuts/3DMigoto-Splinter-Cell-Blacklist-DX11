// ShadowNet - Block colour

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	o0.xyzw = v0.xyzw;
	o1.xy = float2(1,-1) * v0.xy;
	
	float4 tex_filter = IniParams.Load(int2(2,0));
	
	if (tex_filter.w)
	{
		stereo_unfix(o0);
		
		if (tex_filter.x > 0)
			o0.x -= separation * (tex_filter.z * tex_filter.x - tex_filter.z*5)*50;
	}
	return;
}
