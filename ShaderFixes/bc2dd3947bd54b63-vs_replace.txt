#include "CB/VS_Scaleform.hlsl"
#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	float4 v2 : COLOR0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0,
	out float4 o2 : COLOR0)
{
	o0 = mul(v0, sf.pos);
	o1 = v1;
	o2 = v2.zyxw;
	
	stereo_unfix(o0);
	if (texture_filter > 0)
		gui_element(o0,texture_filter,50);
	
	return;
}
