// GUI - Block Colour

cbuffer CB_VS_Scaleform : register(b7)
{
	float4 mvp[8] : packoffset(c0);
}

#include "3Dmigoto.hlsl"

void main(
	int4 v0 : POSITION0,
	float4 v1 : COLOR0,
	float4 v2 : COLOR1,
	out float4 o0 : SV_Position0,
	out float4 o1 : COLOR0,
	out float4 o2 : COLOR1)
{
	float4 r0;

	r0.xyzw = (int4)v0.xyzw;
	o0.x = dot(r0.xyzw, mvp[0].xyzw);
	o0.y = dot(r0.xyzw, mvp[1].xyzw);
	o0.zw = r0.zw;
	o1.xyzw = v1.zyxw;
	o2.xyzw = v2.zyxw;
		
	if (!rt_res_size_match()) // Render Target
		return;
	float layer = 2;
	
	if (o0.x < -0.77 && o0.x > -0.81 && ((o0.y > -0.399 && o0.y < -0.35) || (o0.y > 0.35 && o0.y < 0.395)) && texture_filter == 0) // Menu Vertical Bar
	{
		o0.x -= 0.11;
		layer = 4;
	}
	else if (o0.y < -0.7) // ShadowNet Back
	{
		layer = 5;
	}
	else if (all(v1.xyz == float3(0,0,0)))
	{
		layer = 4;
	}
	
	gui_element(o0,layer,0.002);
	
	return;
}
