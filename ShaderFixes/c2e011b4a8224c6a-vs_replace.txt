// Lighting - Textured

#include "CB/PerCamera.hlsl"
#include "CB/PerObject.hlsl"

#include "3Dmigoto.hlsl"
#include "loadout.hlsl"

void main(
	float4 v0 : POSITION0,
	int4 v1 : TEXCOORD0,
	int2 v2 : TEXCOORD1,
	float4 v3 : NORMAL0,
	float4 v4 : TANGENT0,
	float4 v5 : BINORMAL0,
	out float4 o0 : SV_Position0,
	out float4 o1 : TEXCOORD0)
{
	float4 r0,r1;

	r0.xyz = gDrawScale.xyz * v0.xyz;
	r0.w = v0.w;
	r1.xyz = mul(r0, gLocalToWorld);
	r1.w = 1;
	r0 = mul(r1, gWorldToProj);
	o0 = reorientate(r0);
	o1.zw = r0.zw;
	r0.xy = (int2)v1.xy;
	o1.xy = 0.00048828125 * r0.xy;
	return;
}
