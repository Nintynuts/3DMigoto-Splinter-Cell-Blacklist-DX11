// GUI - HUD + Images 1

cbuffer CB_VS_Scaleform : register(b7)
{
	float4 mvp[8] : packoffset(c0);
}

#include "CB_PerCamera.hlsl"

#include "3Dmigoto.hlsl"

#include "goal_boundaries.hlsl"

void main(
	int4 v0 : POSITION0,
	uint vID : SV_VertexID,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	float4 r0;
		
	r0 = (int4)v0;
	o0.x = dot(r0, mvp[0]);
	o0.y = dot(r0, mvp[1]);
	o0.zw = r0.zw;
	o1.x = dot(r0, mvp[6]);
	o1.y = dot(r0, mvp[7]);
	
	if (!rt_res_size_match()) // Render Target
		return;
	
	o0.x += depth[0] * -eye;
	if (texture_filter > -1)
	{	
		float layer = 4;
	
		if (o0.x == 0 || o0.x == 1 || o0.y == 0 || o0.y == 1)
			layer = 1; // Background Map
		else if (texture_filter > 0)
			layer = texture_filter;

		gui_element(o0, layer, 0.002);
	}
	
	return;
}
