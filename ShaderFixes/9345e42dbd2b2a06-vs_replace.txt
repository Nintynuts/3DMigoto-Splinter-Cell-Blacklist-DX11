// Lighting - Shared

#include "CB_PerCamera.hlsl"

cbuffer CB_PerObject : register(b9)
{
	float4x3 gLocalToWorld : packoffset(c0);
	float4 gDrawScale : packoffset(c3);
	float4 cId : packoffset(c4);
	float4 cSHParams[7] : packoffset(c5);
	float cAOID : packoffset(c12);
	bool bUseSH : packoffset(c12.y);
}

cbuffer CB_VS_SimpleTransform : register(b3)
{
	float4x4 gSimpleWorldToProj : packoffset(c0);
}

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	int2 v1 : TEXCOORD0,
	out float4 o0 : SV_Position0,
	out float4 o1 : TEXCOORD0,
	out float o2 : TEXCOORD1)
{
	float4 r0,r1;

	r0.xyz = mul(v0, gLocalToWorld);
	r0.w = 1;	
	r1 = mul(r0, gSimpleWorldToProj);

	o0.xyzw = r1.xyzw;	
	
	float is_cutscene = IniParams.Load(int2(3, 0)).x;
	
	if (is_cutscene)
		r1.x += separation * r1.w;
	else
		stereo_fix(r1); 
	
	r0.xy = gViewportSB.zw * r1.ww;
	o2.x = r1.w;
	o1.zw = r1.xy * gViewportSB.xy + r0.xy;
	o1.xy = r1.xy;	 
	return;
}
