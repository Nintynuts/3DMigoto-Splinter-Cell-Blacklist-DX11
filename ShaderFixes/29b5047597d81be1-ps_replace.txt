// Lighting - Omni

#include "CB_PerCamera.hlsl"

#include "CB_PS_PerLightBatch.hlsl"

cbuffer CB_PS_DeferredOmniLight : register(b7)
{
	float4x4 cScreenToWorld : packoffset(c0);
}

SamplerState sDepth_s : register(s0);
SamplerState sNormal_s : register(s1);
SamplerState sSceneAlbedo_s : register(s2);
Texture2D<float4> sDepth : register(t0);
Texture2D<float4> sNormal : register(t1);
Texture2D<float4> sSceneAlbedo : register(t2);

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : SV_Position0,
	float4 v1 : TEXCOORD0,
	float v2 : TEXCOORD1,
	out float4 o0 : SV_Target0)
{
	float4 r0,r1,r2,r3,r4;

	r1 = v1 / v2.x;
	r0.xy = r1.xy;
	r0.zw = 1;
	r1.xy = sDepth.Sample(sDepth_s, r1.zw).xw;
	r0.xyz = r1.x * r0.xyz;
			
	stereo_unfix(r0);
	
	r2.xyz = mul(r0, cScreenToWorld);
	r0.xyz = cViewPos.xyz - r2.xyz;
	r2.xyz = cOmni0.pos.xyz - r2.xyz;
	r0.w = dot(r0.xyz, r0.xyz);
	r0.w = rsqrt(r0.w);
	r3.xyz = normalise(r2.xyz);
	r1.x = dot(r2.xyz, r3.xyz);
	r0.xyz = r0.xyz * r0.w + r3.xyz;
	r0.xyz = normalise(r0.xyz);
	r2 = sNormal.Sample(sNormal_s, r1.zw);
	r4 = sSceneAlbedo.Sample(sSceneAlbedo_s, r1.zw);
	r4 = r4 * r4;
	r2.xyz = r2.xyz * 2 - 1;
	r0.w = 11 * r2.w;
	r0.w = exp2(r0.w);
	r0.w = max(0.00001, r0.w);
	r2.xyz = normalise(r2.xyz);
	r0.x = saturate(dot(r0.xyz, r2.xyz));
	r0.y = dot(r2.xyz, r3.xyz);
	r0.x = log2(r0.x);
	r0.x = r0.w * r0.x;
	r0.z = 2 + r0.w;
	r0.x = exp2(r0.x);
	r0.x = r0.z * r0.x;
	r0.z = r0.y * 0.5 + 0.5;
	r0.y = saturate(r0.y);
	r0.z = r0.z * r0.z;
	r0.z = r0.z * r0.z - r0.y;
	r0.y = r1.y * r0.z + r0.y;
	r0.z = saturate(r0.y);
	r1.yz = saturate(r1.x * cOmni0.atten.xy + cOmni0.atten.zw);
	r0.w = cmp(r1.x < cOmni0.pos.w);
	r1.xy = -r1.yz * r1.yz + 1;
	r1.xy = r1.xy * r1.xy;
	r1.z = dot(cOmni0.mask, 1);
	r1.xy = r1.xy * r1.z;
	r0.z = r1.y * r0.z;
	r0.y = r1.x * r0.y;
	r1.xyz = cOmni0.dCol.xyz * r0.y;
	r1.xyz = r0.w ? r1.xyz : 0;
	r0.x = r0.z * r0.x;
	r0.x = 0.125 * r0.x;
	r0.xyz = cOmni0.sCol.xyz * r0.x;
	r0.xyz = r0.w ? r0.xyz : 0;
	r0.w = 8.0/3 * r4.w;
	r0.xyz = r0.xyz * r0.w;
	o0.xyz = r1.xyz * r4.xyz + r0.xyz;
	o0.w = 1;
	return;
}
