// ShadowNet - Decorations

cbuffer CB_VS_Scaleform : register(b7)
{
	float4 mvp[8] : packoffset(c0);
}

#include "3Dmigoto.hlsl"

void main(
	int4 v0 : POSITION0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	float4 r0;
		r0 = (int4)v0;
	o0.x = dot(r0, mvp[0]);
	o0.y = dot(r0, mvp[1]);
	o0.z = dot(r0, mvp[2]);
	o0.w = dot(r0, mvp[3]);
	o1.x = dot(r0, mvp[6]);
	o1.y = dot(r0, mvp[7]);
	
	stereo_unfix(o0);
	float layer = 5;

	if (texture_filter > 0)
		layer = texture_filter;

	if (texture_filter == -3) // Loadout character
		layer = 5;

	if (texture_filter == 2.1)
		layer = 5;

	if (texture_filter == 3.1 && (o0.x < -27500 || o0.x > 27500)) // Large BG map
			layer = 1;
	
	gui_element(o0,layer,50);
		
	return;
}
