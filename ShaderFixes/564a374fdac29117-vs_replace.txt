// ShadowNet - Images

cbuffer CB_VS_Scaleform : register(b7)
{
	float4 mvp[8] : packoffset(c0);
}

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	o0.x = dot(v0.xyzw, mvp[0].xyzw);
	o0.y = dot(v0.xyzw, mvp[1].xyzw);
	o0.z = dot(v0.xyzw, mvp[2].xyzw);
	o0.w = dot(v0.xyzw, mvp[3].xyzw);
	o1.xy = v1.xy;
	
	stereo_unfix(o0);
		
	float4 tex_filter = IniParams.Load(int2(2,0));

	float layer = 4;

	if (tex_filter.x > 0)
		layer = tex_filter.x;
	
	o0.x -= separation * (tex_filter.z * layer - tex_filter.z*5)*50;
	
	return;
}
