// ShadowNet - Images

cbuffer CB_VS_Scaleform : register(b7)
{
	float4 mvp[8] : packoffset(c0);
}

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	float2 v1 : TEXCOORD0,
	out float4 o0 : SV_Position0,
	out float2 o1 : TEXCOORD0)
{
	o0.x = dot(v0, mvp[0]);
	o0.y = dot(v0, mvp[1]);
	o0.z = dot(v0, mvp[2]);
	o0.w = dot(v0, mvp[3]);
	o1.xy = v1.xy;
	
	stereo_unfix(o0);
	float layer = 4;

	if (texture_filter > 0)
		layer = texture_filter;
	
	gui_element(o0,layer,50);
	
	return;
}
