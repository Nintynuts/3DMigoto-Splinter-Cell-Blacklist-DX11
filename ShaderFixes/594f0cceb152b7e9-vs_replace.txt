// Mesh - Bones (POS3,NORM3,VID,TEXCOORDx6 TC5)

#include "CB_PerCamera.hlsl"

#include "CB_PerObject.hlsl"

#define MORPH b5

#include "CB_BoneTransform.hlsl"

cbuffer CB_VS_ColorPalette : register(b4)
{
	float4 cColorPalette[16] : packoffset(c0);
	bool bUseColorPalette : packoffset(c16);
}

#include "3Dmigoto.hlsl"

void main(
	float4 v0 : POSITION0,
	int4 v1 : TEXCOORD0,
	int2 v2 : TEXCOORD1,
	float4 v3 : NORMAL0,
	float4 v4 : TANGENT0,
	float4 v5 : BINORMAL0,
	int4 v6 : BLENDINDICES0,
	uint4 v7 : BLENDWEIGHT0,
	uint v8 : SV_VertexID0,
	out float3 o0 : POSITION_X0,
	out float3 o1 : NORMAL0,
	out uint o2 : VERTEXID0,
	out float4 o3 : TEXCOORD0,
	out float4 o4 : TEXCOORD1,
	out float4 o5 : TEXCOORD2,
	out float4 o6 : TEXCOORD3,
	out float4 o7 : TEXCOORD4,
	out float4 o8 : TEXCOORD5)
{
// Needs manual fix for instruction:
// unknown dcl_: dcl_input_sgv v8.x, vertex_id
	float4 r0,r1,r2,r3,r4,r5,r6,r7;
	
	float3x4 BonesMtx = get_bones_mtx(v6, v7);

	r2.x = gForces.z * v4.w;
	r2.y = v5.w * gForces.w + gForces.x;
	r2.yz = v0.xy * gForces.y + r2.y;
	r2.y = sin(r2.y);
	r4.x = r2.y * r2.x + v0.x;
	r2.y = cos(r2.z);
	r4.y = r2.y * r2.x + v0.y;
	r4.z = v0.z;
	r2.xyz = gDrawScale.xyz * r4.xyz;
	r2.w = v0.w;
	r4.xyz = mul(BonesMtx, r2);
	r2.xyz = -cViewPos_VS.xyz + r4.xyz;
	r5.xyz = mul(r2.xyz, gWorldToView);
	r2.xyz = gViewViewPos.xyz + r5.xyz;
	o4.w = min(64000, r2.z);
	r5.xy = (int2)v1.xy;
	r5.zw = (int2)v2.xy;
	o3 = 0.00048828125 * r5;
	r5.xyz = v4.zyx * 2 - 1;
	r6.xyz = mul(BonesMtx, r5.xyz);
	r5.xyz = v3.zyx * 2 - 1;
	r7.xyz = mul(BonesMtx, r5.xyz);
	r5.xyz = v5.zyx * 2 - 1;
	r0.xyz = mul(BonesMtx, r5.xyz);
	o5.xyz = mul(r6.xyz, gLocalToWorld);
	o6.xyz = mul(r0.xyz, gLocalToWorld);
	r0.xyz = mul(r7.xyz, gLocalToWorld);
	if (bUseColorPalette != 0) {
		r0.w = 255 * v3.w;
		r0.w = (int)r0.w;
		o8 = cColorPalette[r0.w];
	} else {
		o8 = 1;
	}
	r1.xyz = mul(r0.xyz, gWorldToView);
	o1.xyz = normalise(r1.xyz);
	o4.xyz = r4.xyz;
	o5.w = 0;
	o6.w = 0;
	o7.xyz = r0.xyz;
	o7.w = 0;
	o0.xyz = r2.xyz;
	o2.x = v8.x;
	
	//correct_orientation(o0);
	return;
}
